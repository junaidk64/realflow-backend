# RealFlow Backend — Full Test Suite

Run a comprehensive test suite against the running RealFlow backend covering **functionality**, **security**, and **performance**.

## Arguments
- `--url <base>` — override base URL (default: `http://localhost:5000`)
- `--token <jwt>` — provide a valid JWT access token for authenticated tests
- `--only <category>` — run only one suite: `health`, `auth`, `leads`, `workflows`, `templates`, `email`, `gmail`, `smtp`, `settings`, `notifications`, `users`, `admin`, `webhooks`, `security`, `performance`, `ratelimit`
- `--skip-slow` — skip concurrent stress tests

## Instructions

### Step 1 — Parse Arguments
Extract `--url`, `--token`, `--only`, `--skip-slow` from the arguments.
- `BASE_URL` = provided `--url` or `http://localhost:5000`
- `AUTH_TOKEN` = provided `--token` or empty string
- `ONLY` = provided `--only` or `all`
- `SKIP_SLOW` = `true` if `--skip-slow` flag present, else `false`

### Step 2 — Resolve Auth Token

If `--token` was not provided, check if `.env` contains `TEST_JWT`. If still empty, inform the user:
> No auth token provided. Authenticated endpoint tests will be skipped. Use `--token <your-jwt>` to run full tests. Get your token from browser localStorage (`accessToken` key) after logging in.

### Step 3 — Write and Run the Test Script

Write the file `tests/run-backend-tests.mjs` substituting the four constants, then run `node tests/run-backend-tests.mjs`.

Replace these placeholders in the template below before writing:
- `__BASE_URL__` → actual base URL
- `__AUTH_TOKEN__` → actual token (or empty string)
- `__ONLY__` → the only-filter value or `all`
- `__SKIP_SLOW__` → `true` or `false`

// ─── Helpers ──────────────────────────────────────────────────────────────────

const results = { pass: 0, fail: 0, skip: 0, warnings: [], items: [] };

function colorize(code, text) {
  const colors = { green: '\x1b[32m', red: '\x1b[31m', yellow: '\x1b[33m', cyan: '\x1b[36m', bold: '\x1b[1m', reset: '\x1b[0m', dim: '\x1b[2m' };
  return `${colors[code] || ''}${text}${colors.reset}`;
}

function pass(name, detail = '') {
  results.pass++;
  results.items.push({ status: 'PASS', name, detail });
  console.log(`  ${colorize('green', '✓')} ${name}${detail ? colorize('dim', '  ' + detail) : ''}`);
}

function fail(name, detail = '') {
  results.fail++;
  results.items.push({ status: 'FAIL', name, detail });
  console.log(`  ${colorize('red', '✗')} ${name}${detail ? '  ' + colorize('red', detail) : ''}`);
}

function skip(name, reason = '') {
  results.skip++;
  console.log(`  ${colorize('yellow', '○')} ${name}${reason ? colorize('dim', '  (skipped: ' + reason + ')') : ''}`);
}

function warn(msg) {
  results.warnings.push(msg);
  console.log(`  ${colorize('yellow', '⚠')}  ${msg}`);
}

function section(title) {
  console.log(`\n${colorize('bold', colorize('cyan', '┌─ ' + title + ' ' + '─'.repeat(Math.max(0, 55 - title.length))))}`);
}

async function req(method, path, opts = {}) {
  const url = `${BASE_URL}${path}`;
  const headers = { 'Content-Type': 'application/json', ...opts.headers };
  if (AUTH_TOKEN && opts.auth !== false) headers['Authorization'] = `Bearer ${AUTH_TOKEN}`;
  const start = performance.now();
  try {
    const res = await fetch(url, {
      method,
      headers,
      body: opts.body ? JSON.stringify(opts.body) : undefined,
      signal: AbortSignal.timeout(opts.timeout || 10000),
    });
    const ms = Math.round(performance.now() - start);
    let data;
    const ct = res.headers.get('content-type') || '';
    try { data = ct.includes('json') ? await res.json() : await res.text(); } catch { data = null; }
    return { status: res.status, data, ms, headers: res.headers };
  } catch (err) {
    const ms = Math.round(performance.now() - start);
    return { status: 0, data: null, ms, error: err.message };
  }
}

// assertStatus: treats unexpected 429 as skip (rate-limited), not failure
function assertStatus(r, expected, name) {
  const detail = `HTTP ${r.status} in ${r.ms}ms`;
  if (r.status === expected) { pass(name, detail); return; }
  if (r.status === 429 && expected !== 429) { skip(name, `rate limited (429) — window resets every 15min`); return; }
  fail(name, `expected ${expected}, got ${r.status} (${r.ms}ms)`);
}

function assert(condition, name, detail = '') {
  if (condition) pass(name, detail); else fail(name, detail);
}

// safeAssert: skips on 429 instead of failing
function safeAssert(r, condition, name, detail = '') {
  if (r.status === 429) { skip(name, 'rate limited (429)'); return; }
  assert(condition, name, detail);
}

function assertFast(r, name, threshold = 500) {
  if (r.ms <= threshold) pass(name, `${r.ms}ms`);
  else warn(`${name} — slow response ${r.ms}ms (threshold ${threshold}ms)`);
}

    assert(hdrs.get('x-content-type-options') === 'nosniff', 'x-content-type-options: nosniff header present');
    assert(!!hdrs.get('x-frame-options'), 'x-frame-options header present', hdrs.get('x-frame-options'));
    assert(!!hdrs.get('x-xss-protection') || !!hdrs.get('content-security-policy'), 'XSS protection header present');
  }

  const r1 = await fetch(`${BASE_URL}/api/auth/profile`, {
    method: 'GET',
    headers: { 'Authorization': `Bearer ${AUTH_TOKEN || 'fake'}`, 'Origin': 'https://evil.com' }
  });
  const corsHeader = r1.headers.get('access-control-allow-origin');
  assert(corsHeader !== 'https://evil.com' && corsHeader !== '*', 'CORS: evil.com origin rejected or no wildcard', corsHeader || 'none');

  const injPayloads = ['1;DROP TABLE users', '{"$ne":null}', '{"$gt":""}', '<script>alert(1)</script>', "' OR '1'='1"];
  for (const payload of injPayloads) {
    const r = await req('GET', `/api/leads?status=${encodeURIComponent(payload)}`, { auth: false });
    safeAssert(r, r.status === 401 || r.status === 400 || r.status === 200, `Injection payload in query → no 5xx`, payload.slice(0, 30));
  }

  if (AUTH_TOKEN) {
    const r2 = await req('POST', '/api/templates', { body: { name: '<script>alert("XSS")</script>', htmlContent: '<img src=x onerror=alert(1)>', businessType: 'general' } });
    safeAssert(r2, r2.status === 400 || r2.status === 201 || r2.status === 200, 'XSS payload in template body → safe', `${r2.status}`);
    const id = r2.data?._id || r2.data?.template?._id;
    if (id) await req('DELETE', `/api/templates/${id}`);
  }

  const bigBody = JSON.stringify({ data: 'x'.repeat(11 * 1024 * 1024) });
  try {
    const r3 = await fetch(`${BASE_URL}/api/templates`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${AUTH_TOKEN || 'fake'}` },
      body: bigBody,
      signal: AbortSignal.timeout(10000)
    });
    safeAssert({ status: r3.status }, r3.status === 413 || r3.status === 400, 'Request body > 10MB → 413', `${r3.status}`);
  } catch (e) {
    pass('Large payload rejected (connection refused/timeout)', e.message?.slice(0, 50));
  }

  const traversalPaths = ['../../../etc/passwd', '..%2F..%2F..%2Fetc%2Fpasswd'];
  for (const p of traversalPaths) {
    const r = await req('GET', `/api/leads/${encodeURIComponent(p)}`, { auth: false });
    safeAssert(r, r.status !== 200, `Path traversal "${p.slice(0, 25)}" → blocked`, `${r.status}`);
  }

  const r4 = await req('DELETE', '/api/auth/profile', { auth: false });
  safeAssert(r4, r4.status === 404 || r4.status === 405, 'DELETE /api/auth/profile (wrong method) → 404/405', `${r4.status}`);

  const r5 = await req('GET', '/api/leads', { headers: { 'Authorization': 'Bearer fake\r\nX-Admin: true' }, auth: false });
  assert(r5.status === 401 || r5.status === 400 || r5.status === 0 || r5.status === 429, 'Header injection → rejected (401/400/TypeError/429)', `${r5.status || r5.error?.slice(0, 40)}`);

  if (AUTH_TOKEN) {
    const r6 = await req('GET', '/api/leads/507f1f77bcf86cd799439011');
    safeAssert(r6, r6.status === 404, 'IDOR: accessing other user\'s lead → 404', `${r6.status}`);
  }

  const r7 = await req('GET', '/api/nonexistent-endpoint-xyz', { auth: false });
  safeAssert(r7, r7.status === 404, 'Unknown route → 404', `got ${r7.status}`);

  const r8 = await req('POST', '/api/leads/stats', { auth: false });
  safeAssert(r8, r8.status === 404 || r8.status === 405 || r8.status === 401, 'POST to GET-only → 404/405/401', `${r8.status}`);
}

async function testPerformance() {
  section('PERFORMANCE');
  if (!AUTH_TOKEN) console.log('  Note: Running unauthenticated performance tests only.');

  const endpoints = [
    { method: 'GET', path: '/health', auth: false, threshold: 150, label: 'Health check' },
    ...(AUTH_TOKEN ? [
      { method: 'GET', path: '/api/leads', auth: true, threshold: 500, label: 'List leads' },
      { method: 'GET', path: '/api/leads/stats', auth: true, threshold: 600, label: 'Lead stats' },
      { method: 'GET', path: '/api/workflows', auth: true, threshold: 500, label: 'List workflows' },
      { method: 'GET', path: '/api/templates', auth: true, threshold: 400, label: 'List templates' },
      { method: 'GET', path: '/api/notifications', auth: true, threshold: 400, label: 'Notifications' },
      { method: 'GET', path: '/api/settings', auth: true, threshold: 300, label: 'Settings' },
      { method: 'GET', path: '/api/gmail/status', auth: true, threshold: 400, label: 'Gmail status' },
      { method: 'GET', path: '/api/email/provider-status', auth: true, threshold: 400, label: 'Email provider status' },
      { method: 'GET', path: '/api/auth/profile', auth: true, threshold: 300, label: 'Auth profile' },
    ] : [])
  ];

  for (const ep of endpoints) {
    const r = await req(ep.method, ep.path, { auth: ep.auth && !!AUTH_TOKEN });
    const label = `${ep.method} ${ep.path} — ${ep.label}`;
    if (r.status === 0) { fail(label, `Connection error: ${r.error}`); continue; }
    if (r.status === 429) { skip(label, `rate limited (429)`); continue; }
    if (r.ms <= ep.threshold) pass(label, `${r.ms}ms (threshold ${ep.threshold}ms)`);
    else if (r.ms <= ep.threshold * 2) warn(`${label} — ${r.ms}ms (above ${ep.threshold}ms threshold)`);
    else fail(label, `${r.ms}ms (threshold ${ep.threshold}ms) — too slow`);
  }

  if (SKIP_SLOW === 'true') { skip('Concurrent stress tests', '--skip-slow flag set'); return; }

  console.log('\n  Concurrent stress: 20 parallel /health requests...');
  const start = performance.now();
  const concurrent = await Promise.all(Array.from({ length: 20 }, () => req('GET', '/health', { auth: false })));
  const totalMs = Math.round(performance.now() - start);
  const allOk = concurrent.filter(r => r.status === 200 || r.status === 429).length;
  const slowest = Math.max(...concurrent.map(r => r.ms));
  assert(allOk >= 18, `20 concurrent requests: ${allOk}/20 got a response`, `${allOk}/20`);
  assert(slowest < 2000, `Slowest of 20 concurrent < 2000ms`, `${slowest}ms`);
  pass(`20 concurrent requests wall time`, `${totalMs}ms`);

  if (AUTH_TOKEN) {
    console.log('  Concurrent stress: 10 parallel authenticated /api/leads requests...');
    const start2 = performance.now();
    const concurrent2 = await Promise.all(Array.from({ length: 10 }, () => req('GET', '/api/leads?limit=5')));
    const totalMs2 = Math.round(performance.now() - start2);
    const allOk2 = concurrent2.filter(r => r.status === 200 || r.status === 429).length;
    const slowest2 = Math.max(...concurrent2.map(r => r.ms));
    assert(allOk2 >= 8, `10 concurrent /api/leads: ${allOk2}/10 responded`);
    assert(slowest2 < 3000, `Slowest concurrent lead request < 3000ms`, `${slowest2}ms`);
    pass(`10 concurrent auth requests wall time`, `${totalMs2}ms`);
  }
}

// ─── Rate Limiting ─────────────────────────────────────────────────────────────

async function testRateLimiting() {
  section('RATE LIMITING');

  const precheck = await req('GET', '/health', { auth: false });
  if (precheck.status === 429) {
    warn('Already rate limited before burst test — skipping burst to avoid false results');
    skip('Auth rate limit burst', 'pre-existing rate limit active');
    skip('Global rate limit burst', 'pre-existing rate limit active');
    pass('Rate limit is active (pre-existing 429 on /health confirms limiter works)');
    return;
  }

  console.log('  Running auth rate limit burst (25 requests)...');
  const authResults = await Promise.all(
    Array.from({ length: 25 }, (_, i) => req('POST', '/api/auth/refresh', { auth: false, body: { refreshToken: `fake-${i}` } }))
  );
  const auth429 = authResults.filter(r => r.status === 429).length;
  assert(auth429 > 0, `Auth rate limiter triggered (${auth429}/25 requests got 429)`, `${25 - auth429} non-429`);

  console.log('  Running global rate limit burst (201 requests on /health)...');
  const apiResults = await Promise.all(Array.from({ length: 201 }, () => req('GET', '/health', { auth: false })));
  const api429 = apiResults.filter(r => r.status === 429).length;
  if (api429 > 0) pass(`Global rate limiter triggered (${api429}/201 requests got 429)`);
  else warn(`Global rate limiter: no 429s in 201 requests — limit may be higher or already reset`);

  const r = await req('GET', '/health', { auth: false });
  if (r.status !== 429) {
    const hasHeaders = r.headers.get('x-ratelimit-limit') || r.headers.get('ratelimit-limit') || r.headers.get('x-rate-limit-limit');
    if (hasHeaders) pass('Rate limit headers present in normal response', hasHeaders);
    else warn('Rate limit headers not present in normal response (may appear only on 429)');
  } else {
    const retryAfter = r.headers.get('retry-after');
    assert(true, 'Got 429 — rate limiting active', retryAfter ? `Retry-After: ${retryAfter}s` : 'no Retry-After header');
  }
}
